Bienvenido al Rincón del UNO

Proyecto para mecatronicos: circuito para control de 2 motores con push buttons

Descripción:

Aprende a pensar usando una placa Arduino, microcontroladores y código en lenguaje C++.

primero que nada, necesitaras:

1 Arduino UNO(o cualquiera que tengas)

22 cables o jumpers

2 motorreductores

1 Circuito Integrado L293D o un puente H

4 push buttons

4 resistencias de 1k ohmnio

1 bateria de 9v(de las cuadradas)

conecta todo como esta en la imagen

luego carga este codigo en Arduino IDE

// BOTONES

int Badelante = 8;

int Batras = 9;

int Bizquierda = 10;

int Bderecha = 11;

// MOTORES

int motor1 = 4; // Motor 1 adelante

int motor11 = 5; // Motor 1 atrás

int motor2 = 6; // Motor 2 adelante

int motor22 = 7; // Motor 2 atrás

void setup() {

pinMode(Badelante, INPUT_PULLUP);

pinMode(Batras, INPUT_PULLUP);

pinMode(Bizquierda, INPUT_PULLUP);

pinMode(Bderecha, INPUT_PULLUP);

pinMode(motor1, OUTPUT);

pinMode(motor11, OUTPUT);

pinMode(motor2, OUTPUT);

pinMode(motor22, OUTPUT);

}

void loop() {

bool adelante = digitalRead(Badelante) == LOW;

bool atras = digitalRead(Batras) == LOW;

bool izquierda = digitalRead(Bizquierda) == LOW;

bool derecha = digitalRead(Bderecha) == LOW;

// Control motor 1

if (adelante) {

digitalWrite(motor1, HIGH);

digitalWrite(motor11, LOW);

} else if (atras) {

digitalWrite(motor1, LOW);

digitalWrite(motor11, HIGH);

} else {

digitalWrite(motor1, LOW);

digitalWrite(motor11, LOW);

}

// Control motor 2

if (izquierda) {

digitalWrite(motor2, HIGH);

digitalWrite(motor22, LOW);

} else if (derecha) {

digitalWrite(motor2, LOW);

digitalWrite(motor22, HIGH);

} else {

digitalWrite(motor2, LOW);

digitalWrite(motor22, LOW);

}

}
