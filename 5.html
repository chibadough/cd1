Proyecto para juniors: medicion de masa por relacion de distancia

Descripción:

Aprende a usar matematicas aplicadas usando una placa Arduino,sensor sultrasonico y código en lenguaje C++.

primero que nada, necesitaras:

1 Arduino UNO(o cualquiera que tengas)

6 cables o jumpers

1 sensor ultrasonico

1 bote o recipiente cilindrico

agua

conecta todo como esta en la imagen

asegurate de tener en cuenta todas las dimensiones del bote o recipiente cilindrico que usaras, para hacer los ajustes al codigo que yo te proporciono

luego carga este codigo en Arduino IDE

#define TRIGGER_PIN 9

#define ECHO_PIN 10

#define BASE_DISTANCE 13.5 // Nueva distancia base de referencia en cm

#define AREA_BASE 402.91 // cm² (área de la base del cilindro)

#define ALTURA_REF 14 // cm (altura que representa 1 kg o 1000 gramos)

#define PESO_REF 1000 // gramos (1 kg)

// Densidad en gramos por cm³

float densidad = PESO_REF / (AREA_BASE * ALTURA_REF);

void setup() {

Serial.begin(9600);

pinMode(TRIGGER_PIN, OUTPUT);

pinMode(ECHO_PIN, INPUT);

}

void loop() {

float distance = getDistance();

float weight = calculateWeightGrams(distance);

Serial.print("Distancia: ");

Serial.print(distance);

Serial.print(" cm, Peso: ");

Serial.print(weight);

Serial.println(" gramos");

delay(5000);

}

// Función para obtener la distancia en cm

float getDistance() {

digitalWrite(TRIGGER_PIN, LOW);

delayMicroseconds(2);

digitalWrite(TRIGGER_PIN, HIGH);

delayMicroseconds(10);

digitalWrite(TRIGGER_PIN, LOW);

long duration = pulseIn(ECHO_PIN, HIGH);

return duration * 0.034 / 2;

}

// Función para calcular el peso en gramos basado en la distancia

float calculateWeightGrams(float distance) {

if (distance > BASE_DISTANCE) {

return 0; // Si la distancia es mayor a 13.5 cm, el peso es 0

} else {

float altura = BASE_DISTANCE - distance; // Diferencia en altura

float volumen = AREA_BASE * altura; // cm³ (volumen del cilindro con la altura actual)

return densidad * volumen; // gramos

}

}
