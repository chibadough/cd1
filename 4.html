Proyecto para rookies: Control de un motor con sensor ultrasonico

Descripción:

Aprende a encender y apagar un motor DC usando una placa Arduino, un sensor ultrasonico código básico en lenguaje C++.

primero que nada, necesitaras:

1 Arduino UNO(o cualquiera que tengas)

10 cables o jumpers

1 motor

1 sensor ultrasonico

conecta todo como esta en la imagen

luego carga este codigo en Arduino IDE

// Definir pines del sensor ultrasonico

const int trigPin = 9; // Pin TRIG del sensor ultrasónico

const int echoPin = 10; // Pin ECHO del sensor ultrasónico

// Definir pin del motor DC

const int motorPin = 3; // Pin que controla el motor

void setup() {

// Configuración de los pines del sensor ultrasónico

pinMode(trigPin, OUTPUT);

pinMode(echoPin, INPUT);

// Configuración del pin del motor

pinMode(motorPin, OUTPUT);

// Iniciar la comunicación serial

Serial.begin(9600);

}

void loop() {

// Enviar pulso ultrasónico

digitalWrite(trigPin, LOW);

delayMicroseconds(2);

digitalWrite(trigPin, HIGH);

delayMicroseconds(10);

digitalWrite(trigPin, LOW);

// Leer la duración del pulso de retorno

long duration = pulseIn(echoPin, HIGH);

// Calcular la distancia (en cm)

int distance = duration * 0.034 / 2;

// Imprimir la distancia en el monitor serial

Serial.print("Distancia: ");

Serial.print(distance);

Serial.println(" cm");

// Verificar si la distancia es menor o igual a 5 cm

if (distance <= 5) {

digitalWrite(motorPin, HIGH); // Encender motor

} else {

digitalWrite(motorPin, LOW); // Apagar motor

}

delay(100); // Pequeño retraso para estabilizar la lectura

}
